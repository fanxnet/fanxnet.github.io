cd /var/lib/zerotier-one
#zerotier-idtool generate identity.public identity.secret
zerotier-idtool initmoon identity.public > moon.json
修改moon.json,只能IP不能用域名,, 
"stableEndpoints": ["moon_ip/9993"]
   "stableEndpoints": ["116.77.121.106/9993"]

zerotier-idtool genmoon moon.json
mkdir moons.d
mv *.moon moons.d/

拷贝到持久保存路径，
mkdir /etc/config/zero
cp -r /var/lib/zerotier-one/* /etc/config/zero/

修改/etc/config/zerotier，使其从持久目录加载配置文
	config zerotier 'sample_config'
	option enabled '1'
	option config_path '/etc/zerotier'
	option copy_config_path '1'

service zerotier stop
service zerotier restart
